  <script>
    // UUID MIDI BLE officiels
    const SERVICE_UUID = "03b80e5a-ede8-4b33-a751-6ce34ec4c700";
    const MIDI_IO_UUID = "7772e5db-3868-4112-a1a9-f2669d106bf3";

    const bpmValueEl = document.getElementById('bpmValue');
    const statusEl = document.getElementById('status');
    const beatEl = document.getElementById('beat');
    const clientsEl = document.getElementById('clients');
    const logsEl = document.getElementById('logs');
    const connectBtn = document.getElementById('connectBtn');
    const bpmUpBtn = document.getElementById('bpmUp');
    const bpmDownBtn = document.getElementById('bpmDown');
    const tapSyncBtn = document.getElementById('tapSyncBtn');
    const muteBtn = document.getElementById('muteBtn');

    let device = null;
    let server = null;
    let service = null;
    let midiChar = null;

    let bpm = 0;
    let running = false;
    let muted = false;
    let audioCtx = null;
    let masterGainNode = null;
    let beatTimerId = null;
    let intervalMs = 0;
    let tapOriginMs = null;
    let isPlayingBeat = false;

    function log(msg) {
      const ts = new Date().toLocaleTimeString();
      const entry = document.createElement('div');
      entry.className = 'log-entry';
      entry.textContent = `[${ts}] ${msg}`;
      logsEl.appendChild(entry);
      logsEl.scrollTop = logsEl.scrollHeight;
      console.log(msg);
    }

    function updateUI() {
      bpmValueEl.textContent = bpm > 0 ? String(bpm) : '--';
    }

    function setStatus(text) {
      statusEl.textContent = text;
    }

    function updateClients(text) {
      clientsEl.textContent = text;
    }

    function recomputeInterval() {
      intervalMs = bpm > 0 ? (60000 / bpm) : 0;
    }

    function ensureAudio() {
      if (!audioCtx) {
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        masterGainNode = audioCtx.createGain();
        masterGainNode.gain.value = muted ? 0 : 1;
        masterGainNode.connect(audioCtx.destination);
      }
      if (audioCtx.state === "suspended") audioCtx.resume();
    }

    function playClick(accent = false) {
      if (isPlayingBeat) return;
      isPlayingBeat = true;
      ensureAudio();

      const t = audioCtx.currentTime;
      const osc = audioCtx.createOscillator();
      const gain = audioCtx.createGain();

      osc.type = 'square';
      osc.frequency.value = accent ? 1200 : 850;
      gain.gain.setValueAtTime(0.0001, t);
      gain.gain.exponentialRampToValueAtTime(accent ? 0.8 : 0.55, t + 0.001);
      gain.gain.exponentialRampToValueAtTime(0.0001, t + 0.04);

      osc.connect(gain);
      gain.connect(masterGainNode);
      osc.start(t);
      osc.stop(t + 0.05);

      setTimeout(() => { isPlayingBeat = false; }, 60);
    }

    function pulseVisual() {
      beatEl.classList.remove('pulsing');
      void beatEl.offsetWidth;
      beatEl.classList.add('pulsing');
      setTimeout(() => beatEl.classList.remove('pulsing'), 180);
    }

    function scheduleNextBeat() {
      if (!running || bpm <= 0 || !tapOriginMs) return;
      const now = performance.now();
      const elapsed = now - tapOriginMs;
      const nextBeatAtMs = tapOriginMs + (Math.floor(elapsed / intervalMs) + 1) * intervalMs;
      const delay = Math.max(1, nextBeatAtMs - now);
      const accent = (Math.floor(elapsed / intervalMs) % 4) === 0;

      clearTimeout(beatTimerId);
      beatTimerId = setTimeout(() => {
        playClick(accent);
        pulseVisual();
        scheduleNextBeat();
      }, delay);
    }

    function startMetronome() {
      if (bpm <= 0) return;
      ensureAudio();
      recomputeInterval();
      running = true;
      if (!tapOriginMs) tapOriginMs = performance.now();
      setStatus('En lecture');
      log('Métro démarré');
      scheduleNextBeat();
    }

    function stopMetronome() {
      running = false;
      clearTimeout(beatTimerId);
      beatTimerId = null;
      setStatus('Arrêt');
    }

function handleMidiPacket(dataView) {
  const bytes = new Uint8Array(dataView.buffer, dataView.byteOffset, dataView.byteLength);
  if (bytes.length < 2) return;

  const type = bytes[0];
  const value = bytes[1];

  if (type === 0x01) {
    bpm = value;
    recomputeInterval();
    updateUI();
    log(`BPM reçu: ${bpm}`);
    tapOriginMs = performance.now();
    startMetronome();
  } else if (type === 0x02) {
    log(`Beat reçu: ${value}`);
  } else if (type === 0x03) {
    log(`Timecode reçu: ${Array.from(bytes.slice(1)).join(",")}`);
  } else if (type === 0x04) {
    updateClients(`Connectés: ${value}`);
  }
}

async function connectBLE() {
  try {
    setStatus('Sélection ESP32...');

    device = await navigator.bluetooth.requestDevice({
      filters: [{ services: [SERVICE_UUID] }],
      optionalServices: [SERVICE_UUID]
    });

    log(`Connecté: ${device.name}`);

    server = await device.gatt.connect();
    service = await server.getPrimaryService(SERVICE_UUID);
    midiChar = await service.getCharacteristic(MIDI_IO_UUID);
    if (!midiChar) {
      throw new Error("Caractéristique MIDI introuvable");
    }

    const initData = await midiChar.readValue();
    const initBytes = new Uint8Array(initData.buffer, initData.byteOffset, initData.byteLength);
    log(`Raw BPM data: [${Array.from(new Uint8Array(initData.buffer)).join(",")}]`);

    handleMidiPacket(initData);
    ensureAudio();
    if (bpm > 0) {
      tapOriginMs = performance.now();
      startMetronome();
    }

    await midiChar.startNotifications();
    midiChar.addEventListener("characteristicvaluechanged", (event) => {
      handleMidiPacket(event.target.value);
    });

    device.addEventListener("gattserverdisconnected", () => {
      log("Déconnecté");
      stopMetronome();
      bpm = 0;
      updateUI();
      setStatus("Non connecté");
    });

    setStatus("Prêt");
  } catch (err) {
    log(`Erreur: ${err.message}`);
    setStatus("Erreur");
  }
}

connectBtn.onclick = connectBLE;

connectBtn.addEventListener('click', async () => {
  ensureAudio();
  await connectBLE();
});

bpmUpBtn.onclick = async () => {
  if (!midiChar) return;
  const value = Math.min(300, bpm + 1);
  await midiChar.writeValueWithoutResponse(new Uint8Array([0x01, value]));
};
bpmDownBtn.onclick = async () => {
  if (!midiChar) return;
  const value = Math.max(20, bpm - 1);
  await midiChar.writeValueWithoutResponse(new Uint8Array([0x01, value]));
};
tapSyncBtn.onclick = () => {
  tapOriginMs = performance.now();
  log("Tap synchro");
  if (running) scheduleNextBeat();
};
muteBtn.onclick = () => {
  muted = !muted;
  if (masterGainNode && audioCtx) {
    masterGainNode.gain.setValueAtTime(muted ? 0 : 1, audioCtx.currentTime);
  }
  muteBtn.textContent = muted ? "Unmute" : "Mute";
  log(muted ? "Son coupé" : "Son réactivé");
};

updateUI();
log("Page chargée - connectez BLE");
</script>
